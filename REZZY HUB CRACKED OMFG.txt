    local Character = LP.Character or LP.CharacterAdded:Wait()
    local RootPart = Character:WaitForChild("HumanoidRootPart")
    local myPlot = Character:FindFirstAncestorOfClass("Model")

        task.wait(0.12)
        Character = c
        RootPart = c:WaitForChild("HumanoidRootPart")
        myPlot = c:FindFirstAncestorOfClass("Model")
    end)

    local function getBasePart(p) 
        local par = p.Parent
        if par:IsA("BasePart") then return par end
        if par:IsA("Attachment") and par.Parent:IsA("BasePart") then return par.Parent end
        if par:IsA("Model") and par.PrimaryPart then return par.PrimaryPart end
        return nil
    end

    local function isCandidate(prompt)
        if not prompt.Enabled or prompt.ActionText ~= "Steal" then
            return false 
        end
        if math.abs((prompt.HoldDuration or 0) - TARGET_HOLD_DURATION) > 0.1
            or math.abs((prompt.MaxActivationDistance or 0) - TARGET_MAX_DISTANCE) > 1 then
            return false 
        end

        local part = getBasePart(prompt)
        if not part or not RootPart then return false end

        local plotsContainer = Workspace:FindFirstChild("Plots")
        if plotsContainer then
            local plot = part:FindFirstAncestorWhichIsA("Model")
            if plot and plot.Parent == plotsContainer and plot == myPlot then
                return false 
            end
        end


        local dist = (part.Position - RootPart.Position).Magnitude
        return dist <= TARGET_MAX_DISTANCE + 0.5, part, dist
    end

    local function findClosest()
        local best, bestDist = nil, math.huge
        for _, v in ipairs(Workspace:GetDescendants()) do
            if v:IsA("ProximityPrompt") then
                local ok, part, dist = isCandidate(v)
                if ok and dist < bestDist then
                    best = v
                    bestDist = dist
                end
            end
        end
        return best
    end

local function parseTime(text)
    if not text or text == "" then return 0 end
    text = text:lower():gsub("%s", "")
    local value, unit = text:match("(%d+)([sm]?)")
    value = tonumber(value) or 0
    unit = unit or "s"
    if value == 0 then return 0 end
    if unit == "m" then return value * 60 else return value end
end

local function getPlotAttachPart(plotModel)
    if plotModel:IsA("BasePart") then return plotModel end
    if plotModel.PrimaryPart then return plotModel.PrimaryPart end
    for _, part in pairs(plotModel:GetDescendants()) do
        if part:IsA("BasePart") then return part end
    end
    return nil
end

local function createOrUpdatePlotESP(plot, secondsRemaining)
    local espData = activeESPs[plot]

    if not attachPart then return end
    
    if not espData then
        local screenGui = Instance.new("ScreenGui")
        screenGui.Name = "PlotESP_" .. plot.Name

local function updatePlotESP()
    if not plotESPEnabled then return end
    if tick() - lastUpdate < 0.1 then return end
    lastUpdate = tick()

    local plotsFolder = workspace:FindFirstChild("Plots")
    if not plotsFolder then return end

    for _, plot in pairs(plotsFolder:GetChildren()) do
        local multiplierPart = plot:FindFirstChild("Multiplier")
        if multiplierPart then
            local mainGui = multiplierPart:FindFirstChild("Main")
            if mainGui and mainGui:IsA("BillboardGui") then
                local amountLabel = mainGui:FindFirstChild("Amount")
                if amountLabel and amountLabel:IsA("TextLabel") then
                    if amountLabel.Text == "0x" then
                        removePlotESP(plot)
                        lastTimes[plot] = nil
                        stableCounters[plot] = nil
                    else
                        local remainingLabel
                        for _, obj in pairs(plot:GetDescendants()) do
                            if obj:IsA("TextLabel") and obj.Name == "RemainingTime" then
                                remainingLabel = obj
                                break
                            end
                        end

                        if remainingLabel then
                            local seconds = parseTime(remainingLabel.Text)

                            if seconds > 0 then
                                if lastTimes[plot] == seconds then
                                    stableCounters[plot] = (stableCounters[plot] or 0) + 0.1
                                else
                                    stableCounters[plot] = 0
                                    lastTimes[plot] = seconds
                                end

                                if stableCounters[plot] <= 1 then
                                    createOrUpdatePlotESP(plot, seconds)
                                else
                                    removePlotESP(plot)
                                end
                            else
                                removePlotESP(plot)
                                lastTimes[plot] = nil
                                stableCounters[plot] = nil
                            end
                        else
                            removePlotESP(plot)
                            lastTimes[plot] = nil
                            stableCounters[plot] = nil
                        end
                    end
                else
                    removePlotESP(plot)
                    lastTimes[plot] = nil
                    stableCounters[plot] = nil
                end
            else
                removePlotESP(plot)
                lastTimes[plot] = nil
                stableCounters[plot] = nil
            end
        else
            removePlotESP(plot)
            lastTimes[plot] = nil
            stableCounters[plot] = nil
        end
    end
end

local multipliers = { K = 1e3, M = 1e6, B = 1e9, T = 1e12 }

local function parseGenerationValue(text)
    local num, suffix = text:match("([%d%.]+)%s*([KMBT]?)")
    -- Aseguramos la conversión y aplicación del multiplicador
    return num and tonumber(num) * (multipliers[suffix] or 1) or nil 
end

local function scanAndHighlightTopBrainrot()
    if not brainrotESPEnabled then 
        clearOldHighlights()
        return 
    end
    
    local plots = workspace:FindFirstChild('Plots')
    if not plots then return end

    -- 1. LIMPIAR HIGHLIGHTS ANTERIORES CADA VEZ
    clearOldHighlights()
    
    local myPlot = LP.Character and LP.Character:FindFirstAncestorOfClass("Model") or nil
    local animalData = {}

    -- 2. ESCANEO ROBUSTO: BUSCAR EN TODAS PARTES DENTRO DE LAS PARCELAS
    for _, plot in pairs(plots:GetChildren()) do
        if plot == myPlot then continue end
        
        for _, obj in pairs(plot:GetDescendants()) do
            -- Buscamos el objeto que contenga el BillboardGui del Animal
            local AnimalOverhead = obj:FindFirstChild('AnimalOverhead') 
            
            -- Si no lo encontramos directamente, chequeamos si es un apego (Attachment) con el GUI dentro.
            if not AnimalOverhead and obj:IsA("Attachment") then
                 AnimalOverhead = obj:FindFirstChild('AnimalOverhead', true) -- Busca recursivamente si es necesario
            end

            if AnimalOverhead and AnimalOverhead:IsA("BillboardGui") then
                local DisplayName = AnimalOverhead:FindFirstChild('DisplayName')
                local Generation = AnimalOverhead:FindFirstChild('Generation')
                local Rarity = AnimalOverhead:FindFirstChild('Rarity')
                
                if DisplayName and Generation and Rarity then
                    local genValue = parseGenerationValue(Generation.Text)
                    if genValue and genValue > 0 then -- Aseguramos que sea un valor válido y positivo
                        -- Intentamos encontrar el modelo base del animal/objeto
                        local model = obj:FindFirstAncestorOfClass('Model') 
                        if model then
                            table.insert(animalData, {
                                model = model,
                                displayName = DisplayName.Text,
                                generation = Generation.Text,
                                genValue = genValue,
                                rarity = Rarity.Text,
                            })
                        end
                    end
                end
            end
        end
    end

    table.sort(animalData, function(a, b) return a.genValue > b.genValue end)

            highlight.FillTransparency = 0.3
            highlight.OutlineTransparency = 0
            highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop 
            highlight.Enabled = true

            local primaryPart = model.PrimaryPart or model:FindFirstChild('HumanoidRootPart')
            
            local billboard = Instance.new('BillboardGui')
            billboard.Parent = primaryPart or model -- Adjuntamos al modelo principal
            billboard.Size = UDim2.new(0, 200, 0, 60)
            billboard.StudsOffset = Vector3.new(0, primaryPart and primaryPart.Size.Y or 5, 0)
            billboard.AlwaysOnTop = true

            local nameLabel = Instance.new('TextLabel')
            nameLabel.Parent = billboard
            nameLabel.Size = UDim2.new(1, 0, 0.5, 0)
            nameLabel.BackgroundTransparency = 1
            nameLabel.Text = topBrainrot.displayName

--========================================================
-- MOTOR DE VELOCIDAD CHILLI (ULTRA RÁPIDO)
--========================================================
local function makeBoostController()
    local self = { conn = nil }
    local step = 1/120 

    function self:stop()
        if self.conn then self.conn:Disconnect(); self.conn = nil end
    end

    function self:start(hum, hrp)
        self:stop()
        local acc = 0
        self.conn = RunService.Heartbeat:Connect(function(dt)
            acc += dt
            if acc < step then return end
            while acc >= step do
                acc -= step
                if not hum or hum.Health <= 0 or not hrp then break end

                local v = hrp.AssemblyLinearVelocity
                local horiz = Vector3.new(v.X, 0, v.Z)
                local md = hum.MoveDirection
                local onGround = hum.FloorMaterial ~= Enum.Material.Air

                if md.Magnitude > 0.01 then
                    -- Lógica de velocidad máxima: WalkSpeed + Boost
                    local targetSpeed = math.clamp(hum.WalkSpeed + config.SpeedBoost_Speed, 0, config.Max_Limit)
                    local targetHoriz = md.Unit * targetSpeed
                    
                    -- Respuesta ultra rápida (14 en suelo, 10 en aire)
                    local resp = onGround and 14 or 10
                    local alpha = 1 - math.exp(-resp * step)
                    local newHoriz = horiz:Lerp(targetHoriz, alpha)
                    
                    hrp.AssemblyLinearVelocity = Vector3.new(newHoriz.X, v.Y, newHoriz.Z)
                else
                    -- Frenado (Decay)
                    local decay = onGround and 18 or 4
                    local factor = math.exp(-decay * step)
                    hrp.AssemblyLinearVelocity = Vector3.new(horiz.X * factor, v.Y, horiz.Z * factor)
                end
            end
        end)
    end
    return self
end

    main.Size = UDim2.fromOffset(240, 110)
    main.Position = UDim2.new(0.1, 0, 0.1, 0)
    main.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
    Instance.new("UICorner", main).CornerRadius = UDim.new(0, 6)
    local mainStroke = Instance.new("UIStroke", main)
    mainStroke.Color = Color3.fromRGB(200, 0, 0)
    mainStroke.Thickness = 2

    -- Título
    local title = Instance.new("TextLabel", main)
    title.Size = UDim2.new(1, 0, 0, 35)
    title.Text = "REZZY SPEEDHACK"
    title.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
    title.BackgroundTransparency = 0.2
    title.TextColor3 = Color3.new(1,1,1)
    title.Font = Enum.Font.GothamBold
    Instance.new("UICorner", title).CornerRadius = UDim.new(0, 6)

    -- Botón X
    local closeBtn = Instance.new("TextButton", title)
    closeBtn.Text = "X"
    closeBtn.Size = UDim2.fromOffset(30, 35)
    closeBtn.Position = UDim2.new(1, -35, 0, 0)
    closeBtn.BackgroundTransparency = 1
    closeBtn.TextColor3 = Color3.new(1,1,1)
    closeBtn.Font = Enum.Font.GothamBold

    -- Botón Activate (Tu diseño)
    local toggle = Instance.new("TextButton", main)
    toggle.Size = UDim2.new(0.85, 0, 0, 40)
    toggle.Position = UDim2.new(0.075, 0, 0.45, 0)
    toggle.BackgroundColor3 = Color3.fromRGB(30, 0, 0)
    toggle.Text = "STATUS: OFF"
    toggle.TextColor3 = Color3.new(1,1,1)
    toggle.Font = Enum.Font.GothamSemibold
    Instance.new("UICorner", toggle)
    local tStroke = Instance.new("UIStroke", toggle)
    tStroke.Color = Color3.fromRGB(100, 0, 0)

    -- Visibilidad y Arrastre
    closeBtn.MouseButton1Click:Connect(function() main.Visible = false; topButton.Visible = true end)
    topButton.MouseButton1Click:Connect(function() main.Visible = true; topButton.Visible = false end)

    local dragging, dragStart, startPos
    title.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true; dragStart = input.Position; startPos = main.Position
            input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)

    -- Lógica del Botón Activate
    toggle.MouseButton1Click:Connect(function()
        config.Move_CFrame = not config.Move_CFrame
        grappleEnabled = config.Move_CFrame
        if config.Move_CFrame then
            toggle.Text = "STATUS: ACTIVE"
            toggle.BackgroundColor3 = Color3.fromRGB(150, 0, 0)
            tStroke.Color = Color3.fromRGB(255, 0, 0)
            equipGrapple()
            if player.Character and player.Character:FindFirstChild("Humanoid") then
                boost:start(player.Character.Humanoid, player.Character.HumanoidRootPart)
            end
        else
            toggle.Text = "STATUS: OFF"
            toggle.BackgroundColor3 = Color3.fromRGB(30, 0, 0)
            tStroke.Color = Color3.fromRGB(100, 0, 0)
            boost:stop()
        end
    end)
end
