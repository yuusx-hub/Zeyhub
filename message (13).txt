-- Logic for Draggable Open Button

    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
        draggingBtn = true 
        dragStartBtn = input.Position 
        startPosBtn = OpenButton.Position 
        clickStartTime = tick() 
    end 
end) 

UserInputService.InputChanged:Connect(function(input) 
    if draggingBtn and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then 
        local delta = input.Position - dragStartBtn 
        OpenButton.Position = UDim2.new(startPosBtn.X.Scale, startPosBtn.X.Offset + delta.X, startPosBtn.Y.Scale, startPosBtn.Y.Offset + delta.Y) 
    end 
end) 

UserInputService.InputEnded:Connect(function(input) 
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
        draggingBtn = false 
    end 
end)

local function AddSlider(parentContainer, sliderText, minVal, maxVal, defaultVal, callbackFunction) 
    local Frame = Instance.new("Frame", parentContainer) 
    Frame.Size = UDim2.new(1, -10, 0, 55) 
    Frame.BackgroundColor3 = Color3.fromRGB(35, 35, 35) 
    Instance.new("UICorner", Frame) 
    Instance.new("UIStroke", Frame).Color = Color3.fromRGB(10, 10, 10) 
    
    local Label = Instance.new("TextLabel", Frame) 
    Label.Text = sliderText .. " : " .. tostring(defaultVal) 
    Label.Size = UDim2.new(1, -20, 0, 20) 
    Label.Position = UDim2.new(0, 10, 0, 5) 
    Label.TextColor3 = Color3.new(1, 1, 1) 
    Label.BackgroundTransparency = 1 
    Label.TextXAlignment = Enum.TextXAlignment.Left 
    Label.Font = Enum.Font.Gotham 
    
    local SliderTrack = Instance.new("Frame", Frame) 
    SliderTrack.Size = UDim2.new(1, -20, 0, 6) 
    SliderTrack.Position = UDim2.new(0, 10, 0, 35) 
    SliderTrack.BackgroundColor3 = Color3.fromRGB(20, 20, 20) 
    Instance.new("UICorner", SliderTrack).CornerRadius = UDim.new(1, 0) 
    
    local SliderFill = Instance.new("Frame", SliderTrack) 
    SliderFill.Size = UDim2.new((defaultVal - minVal) / (maxVal - minVal), 0, 1, 0) 
    SliderFill.BackgroundColor3 = Color3.fromRGB(40, 200, 80) 
    Instance.new("UICorner", SliderFill).CornerRadius = UDim.new(1, 0) 
    
    local SliderBall = Instance.new("TextButton", SliderTrack) 
    SliderBall.Size = UDim2.new(0, 16, 0, 16) 
    SliderBall.Position = UDim2.new((defaultVal - minVal) / (maxVal - minVal), -8, 0.5, -8) 
    SliderBall.BackgroundColor3 = Color3.new(1, 1, 1) 
    SliderBall.Text = "" 
    Instance.new("UICorner", SliderBall).CornerRadius = UDim.new(1, 0) 
    
    local isDraggingSlider = false 
    local function processSliderMovement(input) 
        local clampedPosition = math.clamp((input.Position.X - SliderTrack.AbsolutePosition.X) / SliderTrack.AbsoluteSize.X, 0, 1) 
        local newValue = math.floor(minVal + ((maxVal - minVal) * clampedPosition)) 
        SliderFill.Size = UDim2.new(clampedPosition, 0, 1, 0) 
        SliderBall.Position = UDim2.new(clampedPosition, -8, 0.5, -8) 
        Label.Text = sliderText .. " : " .. tostring(newValue) 
        callbackFunction(newValue) 
    end 
    
    SliderBall.InputBegan:Connect(function(input) 
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
            isDraggingSlider = true 
        end 
    end) 
    
    UserInputService.InputEnded:Connect(function(input) 
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
            isDraggingSlider = false 
        end 
    end) 
    
    UserInputService.InputChanged:Connect(function(input) 
        if isDraggingSlider and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then 
            processSliderMovement(input) 
        end 
    end) 
end

-- Game Logic Helpers (Murder Mystery 2)
local function getPlayerRole(playerTarget) 
    if playerTarget.Backpack:FindFirstChild("Knife") or (playerTarget.Character and playerTarget.Character:FindFirstChild("Knife")) then 
        return "Murderer" 
    end 
    if playerTarget.Backpack:FindFirstChild("Gun") or (playerTarget.Character and playerTarget.Character:FindFirstChild("Gun")) then 
        return "Sheriff" 
    end 
    return "Innocent" 
end 

local function getGunDropLocation() 
    for _, item in pairs(workspace:GetDescendants()) do 
        if item.Name == "GunDrop" then 
            if item:IsA("BasePart") then 
                return item.CFrame 
            elseif item:IsA("Model") and item.PrimaryPart then 
                return item.PrimaryPart.CFrame 
            end 
        end 
    end 
    return nil 
end

    _G.AutoFarmEnabled = isEnabled 
    if isEnabled then 
        autoFarmNoclipConn = RunService.Stepped:Connect(function() 
            if _G.AutoFarmEnabled and LocalPlayer.Character then 
                for _, bodyPart in pairs(LocalPlayer.Character:GetDescendants()) do 
                    if bodyPart:IsA("BasePart") then bodyPart.CanCollide = false end 
                end 
            end 
        end) 
        task.spawn(function() 
            while _G.AutoFarmEnabled do 
                task.wait(0.1) 
                pcall(function() 
                    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then 
                        for _, mapObject in ipairs(workspace:GetDescendants()) do 
                            if not _G.AutoFarmEnabled then 
                                if autoFarmTween then autoFarmTween:Cancel() end break 
                            end 
                            if (mapObject.Name:lower():find("coin") or mapObject.Name:lower():find("gold")) and (mapObject:IsA("BasePart") or mapObject:IsA("MeshPart")) and mapObject.Transparency < 1 then 
                                local distance = (LocalPlayer.Character.HumanoidRootPart.Position - mapObject.Position).Magnitude 
                                local tweenInfo = TweenInfo.new(distance / 15, Enum.EasingStyle.Linear) 
                                autoFarmTween = TweenService:Create(LocalPlayer.Character.HumanoidRootPart, tweenInfo, {CFrame = mapObject.CFrame}) 
                                autoFarmTween:Play() 
                                autoFarmTween.Completed:Wait() 
                                if firetouchinterest then 
                                    firetouchinterest(LocalPlayer.Character.HumanoidRootPart, mapObject, 0) 
                                    task.wait() 
                                    firetouchinterest(LocalPlayer.Character.HumanoidRootPart, mapObject, 1) 
                                end 
                            end 
                        end 
                    end 
                end) 
            end 
        end) 
    else 
        if autoFarmNoclipConn then autoFarmNoclipConn:Disconnect() autoFarmNoclipConn = nil end 
        if autoFarmTween then autoFarmTween:Cancel() autoFarmTween = nil end 
    end 
end) 

local godModeEnabled, deathPosition, godModeConnections = false, nil, {} 
local function applyGodModeRules(character) 
    if not godModeEnabled then return end 
    local humanoid = character:WaitForChild("Humanoid", 5) 
    local rootPart = character:WaitForChild("HumanoidRootPart", 5) 
    if not humanoid or not rootPart then return end 
    
    if deathPosition then 
        task.wait(0.05) 
        rootPart.CFrame = CFrame.new(deathPosition) 
        deathPosition = nil 
    end 
    
    table.insert(godModeConnections, humanoid.Died:Connect(function() 
        if godModeEnabled then 
            deathPosition = rootPart.Position 
            task.wait(0.05) 
            LocalPlayer:LoadCharacter() 
        end 
    end)) 
    
    table.insert(godModeConnections, humanoid.HealthChanged:Connect(function(health) 
        if godModeEnabled and health <= 0 then humanoid.Health = humanoid.MaxHealth end 
    end)) 
    humanoid.Health = humanoid.MaxHealth 
    
    table.insert(godModeConnections, RunService.Heartbeat:Connect(function() 
        if rootPart and rootPart.Position.Y < -100 and godModeEnabled then 
            godModeEnabled = false 
            humanoid.Health = 0 
        end 
    end)) 
end 

AddToggle(TabMain, "God Mode", false, function(isEnabled) 
    godModeEnabled = isEnabled 
    if isEnabled then 
        if LocalPlayer.Character then applyGodModeRules(LocalPlayer.Character) end 
        table.insert(godModeConnections, LocalPlayer.CharacterAdded:Connect(applyGodModeRules)) 
    else 
        for _, connection in ipairs(godModeConnections) do connection:Disconnect() end 
        table.clear(godModeConnections) 
        deathPosition = nil 
    end 
end)

local flingEnabled, flingLoopConnection = false, nil 
AddToggle(TabFling, "Enable Fling", false, function(isEnabled) 
    flingEnabled = isEnabled 
    if flingEnabled then 
        flingLoopConnection = RunService.Stepped:Connect(function() 
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and LocalPlayer.Character:FindFirstChild("Head") then 
                LocalPlayer.Character.Head.CanCollide = false 
                LocalPlayer.Character.HumanoidRootPart.Velocity = Vector3.new(0, 0, 0) 
                LocalPlayer.Character.HumanoidRootPart.RotVelocity = Vector3.new(0, 3500, 0) 
            end 
        end) 
        task.spawn(function() 
            while flingEnabled do 
                task.wait(0.1) 
                for _, otherPlayer in pairs(Players:GetPlayers()) do 
                    if not flingEnabled then break end 
                    if otherPlayer ~= LocalPlayer and otherPlayer.Character and otherPlayer.Character:FindFirstChild("HumanoidRootPart") then 
                        local currentRole = getPlayerRole(otherPlayer) 
                        if selectedFlingTarget == "All Players" or selectedFlingTarget == currentRole then 
                            local targetRoot = otherPlayer.Character.HumanoidRootPart 
                            local timer = 0 
                            while timer < 2 and flingEnabled do 
                                if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then 
                                    LocalPlayer.Character.HumanoidRootPart.CFrame = targetRoot.CFrame 
                                end 
                                task.wait(0.1) 
                                timer = timer + 0.1 
                            end 
                        end 
                    end 
                end 
            end 
        end) 
    else 
        if flingLoopConnection then flingLoopConnection:Disconnect() flingLoopConnection = nil end 
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then 
            LocalPlayer.Character.HumanoidRootPart.Velocity = Vector3.new(0, 0, 0) 
            LocalPlayer.Character.HumanoidRootPart.RotVelocity = Vector3.new(0, 0, 0) 
        end 
    end 
end) 

local touchFlingConn 
AddToggle(TabFling, "Touch Fling", false, function(isEnabled) 
    local moveVelocity = 0.1 
    if isEnabled then 
        touchFlingConn = RunService.Heartbeat:Connect(function() 
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then 
                local rootPart = LocalPlayer.Character.HumanoidRootPart 
                local originalVelocity = rootPart.Velocity 
                rootPart.Velocity = originalVelocity * 10000 + Vector3.new(0, 10000, 0) 
                RunService.RenderStepped:Wait() 
                rootPart.Velocity = originalVelocity 
                RunService.Stepped:Wait() 
                rootPart.Velocity = originalVelocity + Vector3.new(0, moveVelocity, 0) 
                moveVelocity = -moveVelocity 
            end 
        end) 
    else 
        if touchFlingConn then touchFlingConn:Disconnect() touchFlingConn = nil end 
    end 
end)

AddToggle(TabVisuals, "ESP Roles", false, function(isEnabled) 
    _G.ESP_Roles = isEnabled 
    task.spawn(function() 
        while _G.ESP_Roles do 
            task.wait(1) 
            for _, otherPlayer in pairs(Players:GetPlayers()) do 
                if otherPlayer ~= LocalPlayer and otherPlayer.Character then 
                    if not otherPlayer.Character:FindFirstChild("PlayerHighlight") then 
                        local highlight = Instance.new("Highlight", otherPlayer.Character) 
                        highlight.Name = "PlayerHighlight" 
                    end 
                    local currentHighlight = otherPlayer.Character.PlayerHighlight 
                    if otherPlayer.Backpack:FindFirstChild("Knife") or otherPlayer.Character:FindFirstChild("Knife") then 
                        currentHighlight.FillColor = Color3.new(1, 0, 0) -- Red for Murderer
                    elseif otherPlayer.Backpack:FindFirstChild("Gun") or otherPlayer.Character:FindFirstChild("Gun") then 
                        currentHighlight.FillColor = Color3.new(0, 0, 1) -- Blue for Sheriff
                    else 
                        currentHighlight.FillColor = Color3.new(0, 1, 0) -- Green for Innocent
                    end 
                end 
            end 
        end 
        -- Cleanup ESP when turned off
        if not _G.ESP_Roles then 
            for _, otherPlayer in pairs(Players:GetPlayers()) do 
                if otherPlayer.Character and otherPlayer.Character:FindFirstChild("PlayerHighlight") then 
                    otherPlayer.Character.PlayerHighlight:Destroy() 
                end 
            end 
        end 
    end) 
end)

-- Window Dragging Logic
local isDraggingWindow, windowDragStart, windowStartPos 
TopBar.InputBegan:Connect(function(input) 
    if input.UserInputType == Enum.UserInputType.MouseButton1 then 
        isDraggingWindow = true 
        windowDragStart = input.Position 
        windowStartPos = MainFrame.Position 
    end 
end) 

UserInputService.InputChanged:Connect(function(input) 
    if isDraggingWindow and input.UserInputType == Enum.UserInputType.MouseMovement then 
        local delta = input.Position - windowDragStart 
        MainFrame.Position = UDim2.new(windowStartPos.X.Scale, windowStartPos.X.Offset + delta.X, windowStartPos.Y.Scale, windowStartPos.Y.Offset + delta.Y) 
    end 
end) 

UserInputService.InputEnded:Connect(function(input) 
    if input.UserInputType == Enum.UserInputType.MouseButton1 then 
        isDraggingWindow = false 
    end 
end) 

OpenButton.InputEnded:Connect(function(input) 
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
        if tick() - clickStartTime < 0.2 then 
            MainFrame.Visible = not MainFrame.Visible 
        end 
    end 
end) 

CloseBtn.MouseButton1Click:Connect(function() 
    ScreenGui:Destroy() 
end)
