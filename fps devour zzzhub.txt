-- drag
local dragging = false
local dragStart = nil
local startPos = nil

local function updateInput(input)
    local delta = input.Position - dragStart
    frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

titleBar.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateInput(input)
    end
end)

-- ── STATE ──────────────────────────────────────────────
local isRunning = false
local devourTask = nil

local function startDevourer()
    if devourTask then task.cancel(devourTask) end
    devourTask = task.spawn(function()
        while isRunning do
            local char = player.Character
            if char and char:FindFirstChild("Humanoid") then
                local hum = char.Humanoid
                if not char:FindFirstChild(TOOL_NAME) then
                    hum:EquipTool(tool)
                else
                    hum:UnequipTools()
                end
            end
            task.wait(EQUIP_SPEED)
        end
    end)
end

local function stopDevourer()
    isRunning = false
    if devourTask then
        task.cancel(devourTask)
        devourTask = nil
    end
    local char = player.Character
    if char and char:FindFirstChild("Humanoid") then
        char.Humanoid:UnequipTools()
    end
end

button.MouseButton1Click:Connect(function()
    isRunning = not isRunning
    if isRunning then
        button.Text = "fps devourer: ON"
        button.BackgroundColor3 = Color3.fromRGB(0, 140, 0)
        startDevourer()
    else
        button.Text = "fps devourer: OFF"
        button.BackgroundColor3 = Color3.fromRGB(100, 0, 0)
        stopDevourer()
    end
end)

local speeds = {0.07, 0.08, 0.09, 0.1, 0.2, 0.5}
local speedIndex = 4 -- default 0.1
speedLabel.MouseButton1Click:Connect(function()
    speedIndex = (speedIndex % #speeds) + 1
    EQUIP_SPEED = speeds[speedIndex]
    speedLabel.Text = "discord.gg/zzzzhub • Speed: " .. tostring(EQUIP_SPEED)
end)

            TweenService:Create(button, TweenInfo.new(0.3), {BackgroundColor3 = Color3.fromRGB(0, 255, 0)}):Play()

            if isRunning then
                TweenService:Create(button, TweenInfo.new(0.3), {BackgroundColor3 = Color3.fromRGB(0, 140, 0)}):Play()
            end
        end
    end
end)
