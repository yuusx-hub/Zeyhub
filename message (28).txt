local function createSetting(parent, y, text, val, minVal, maxVal, step, isNum)

    rowFrame.Size = UDim2.new(1, 0, 0, 28)
    rowFrame.Position = UDim2.new(0, 0, 0, y)
    rowFrame.BackgroundTransparency = 1
    rowFrame.Parent = parent
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(0.5, 0, 1, 0)
    label.Position = UDim2.new(0, 8, 0, 0)
    label.BackgroundTransparency = 1
    label.Text = text
    label.TextColor3 = Color3.fromRGB(180, 180, 210)
    label.TextSize = 10
    label.Font = Enum.Font.GothamBold
    label.TextXAlignment = Enum.TextXAlignment.Left
    label.Parent = rowFrame
    local valueFrame = Instance.new("Frame")
    valueFrame.Size = UDim2.new(0, 42, 0, 22)
    valueFrame.Position = UDim2.new(0.55, 0, 0, 3)
    valueFrame.BackgroundColor3 = Color3.fromRGB(50, 50, 65)
    valueFrame.BorderSizePixel = 0
    valueFrame.Parent = rowFrame
    local valCorner = Instance.new("UICorner")
    valCorner.CornerRadius = UDim.new(0, 5)
    valCorner.Parent = valueFrame
    local valueLabel = Instance.new("TextLabel")
    valueLabel.Size = UDim2.new(1, 0, 1, 0)
    valueLabel.BackgroundTransparency = 1
    valueLabel.Text = (isNum and tostring(val)) or string.format("%.0f", val)
    valueLabel.TextColor3 = Color3.fromRGB(0, 200, 255)
    valueLabel.TextSize = 10
    valueLabel.Font = Enum.Font.GothamBold
    valueLabel.Parent = valueFrame
    local minus = Instance.new("TextButton")
    minus.Size = UDim2.new(0, 24, 0, 22)
    minus.Position = UDim2.new(0.74, 0, 0, 3)
    minus.BackgroundColor3 = Color3.fromRGB(50, 50, 65)
    minus.Text = "-"
    minus.TextColor3 = Color3.fromRGB(255, 100, 100)
    minus.TextSize = 16
    minus.Font = Enum.Font.GothamBold
    minus.Parent = rowFrame
    local minusCorner = Instance.new("UICorner")
    minusCorner.CornerRadius = UDim.new(0, 5)
    minusCorner.Parent = minus
    local plus = Instance.new("TextButton")
    plus.Size = UDim2.new(0, 24, 0, 22)
    plus.Position = UDim2.new(0.84, 0, 0, 3)
    plus.BackgroundColor3 = Color3.fromRGB(50, 50, 65)
    plus.Text = "+"
    plus.TextColor3 = Color3.fromRGB(100, 255, 100)
    plus.TextSize = 16
    plus.Font = Enum.Font.GothamBold
    plus.Parent = rowFrame
    local plusCorner = Instance.new("UICorner")
    plusCorner.CornerRadius = UDim.new(0, 5)
    plusCorner.Parent = plus
    local function update()
        if isNum then
            valueLabel.Text = tostring(val)
        else
            valueLabel.Text = string.format("%.0f", val)
        end
    end
    minus.MouseButton1Click:Connect(function()
    val = math.max(minVal, val - step)
    if not isNum then
        val = math.floor(val)
    end
    update()
    if active then
        stopLagger()
        startLagger()
    end
end)
    plus.MouseButton1Click:Connect(function()
    val = math.min(maxVal, val + step)
    if not isNum then
        val = math.floor(val)
    end
    update()
    if active then
        stopLagger()
        startLagger()
    end
end)
    return function()
    return val
end
end

end
local function startLagger()
    active = true
    stopFlag = false
    updateUI()
    local threads = getThreads()
    local bursts = getBursts()
    local delay = getDelay() / (1000)
    pcall(function()
    local NetworkClient = game:GetService("NetworkClient")
    if NetworkClient and NetworkClient.SetOutgoingKBPSLimit then
        NetworkClient:SetOutgoingKBPSLimit(1)
    end
end)
    remoteEvent = Instance.new("RemoteEvent")
    remoteFunction = Instance.new("RemoteFunction")
    for i = 1, threads do
        if stopFlag then
            break
        end
        task.spawn(function()
    while not stopFlag do
        pcall(function()
    remoteEvent:FireServer()
    remoteFunction:InvokeServer()
    HttpService:JSONEncode(("0xd4d210c2b40ac628") .. math.random())
    local rrs = ReplicatedStorage:FindFirstChild("RobloxReplicatedStorage")
    if rrs and rrs.SetPlayerBlockList then
        rrs.SetPlayerBlockList:FireServer()
    end
end)
        task.wait(delay)
    end
end)
    end
    for i = 1, bursts do
        if stopFlag then
            break
        end
        task.spawn(function()
    while not stopFlag do
        pcall(function()
    local camera = workspace.CurrentCamera
    if camera then
        camera.FieldOfView = (70) + math.random(-3, 3)
    end
end)
        task.wait(delay * (2))
    end
end)
    end
end
local function stopLagger()
    active = false
    stopFlag = true
    updateUI()
    pcall(function()
    local NetworkClient = game:GetService("NetworkClient")
    if NetworkClient and NetworkClient.SetOutgoingKBPSLimit then
        NetworkClient:SetOutgoingKBPSLimit(0)
    end
end)
    if remoteEvent then
        remoteEvent:Destroy()
    end
    if remoteFunction then
        remoteFunction:Destroy()
    end
end
local function toggleLagger()
    if active then
        stopLagger()
    else
        startLagger()
    end
end
switchClick.MouseButton1Click:Connect(toggleLagger)
textButton.MouseButton1Click:Connect(function()
    stopLagger()
    screenGui:Destroy()
end)
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then
        return
    end
    if (input.KeyCode == Enum.KeyCode.LeftControl) or (input.KeyCode == Enum.KeyCode.RightControl) then
        frame.Visible = not frame.Visible
    end
    if input.KeyCode == Enum.KeyCode.V then
        toggleLagger()
    end
end)
local oldThreads, oldBursts, oldDelay = getThreads(), getBursts(), getDelay()
task.spawn(function()
    while true do
        task.wait(0.5)
        if active then
            local newThreads = getThreads()
            local newBursts = getBursts()
            local newDelay = getDelay()
            if ((newThreads ~= oldThreads) or (newBursts ~= oldBursts)) or (newDelay ~= oldDelay) then
                oldThreads, oldBursts, oldDelay = newThreads, newBursts, newDelay
                stopLagger()
                startLagger()
            end
        end
    end
end)
lp.AncestryChanged:Connect(function()
    if not lp.Parent then
        stopLagger()
    end
end)
